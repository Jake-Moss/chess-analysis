\color{EFD}\EFk{def} \EFf{load\_pgn}(filename: \EFb{str}) -> \EFv{List}[chess.pgn.Game]:
    \EFk{with} \EFb{open}(filename) \EFk{as} pgn:
        games =[]
        \EFk{while} \EFc{True}:
            game = chess.pgn.read\_game(pgn)
            \EFk{if} game \EFk{is} \EFk{not} \EFc{None}:
                games.append(game)
            \EFk{else}:
                \EFk{break}
        \EFk{return} games
\end{Verbatim}
\end{Code}
